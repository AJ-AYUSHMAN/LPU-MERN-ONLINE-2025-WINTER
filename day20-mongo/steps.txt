6. Inside this "dbConfig.js" file,
`
    const mongoose = require("mongoose");

    const connectToDb = async () => {
        try {
            await mongoose.connect("");
            console.log("---------- ✅ Database Connected ---------");
        } catch (err) {
            console.log("-------- ❌ Database NOT Connected -------");
        }
    };

    connectToDb();
`
